## 主动调用其他类的成员

方式一：`Base.f1(self)`，与继承无关

```python
class Base(object):
    def f1(self):
        print('5个功能')

class Foo(Base):
    def f1(self):
        print('3个功能')
        Base.f1(self)
        
obj = Foo()
obj.f1()
```

方式二：按照类的继承顺序找下一个

```python
class Base(object):
    def f1(self):
        print('5个功能')

class Foo(Base):
    def f1(self):
        print('3个功能')
        super().f1()
        
obj = Foo()
obj.f1()
```

```python
class Foo(object):
    def f1(self):
        super().f1()
        print('3个功能')
def Bar(object):
    def f1(self):
        print('6个功能')
class Info(Foo, Bar):
    pass
        
obj = Info()
obj.f1()
```

## 特殊成员

```python
class Foo(object):
    def __init__(self,a1,a2):  # 对数据进行初始化
        print(1)
        self.a1 = a1
        self.a2 = a2
    def __new__(cls,*args,**kwargs):  # 创建一个空的对象
        print(2)
        return object.__new__(cls)  # python内部创建一个当前类的对象，初建时内部是空的
    def __call__(self,*args,**kxargs):
        print(1111,args,kwargs)
    	return 123
    def __getitem__(self,item):
        print(item)
        return 4
    def __setitem__(self, value):
        print(key, value, 111111)
    def __delitem__(self, key):
        print(key)
    def __add__(self, other):
        return self.a1 + other.a2
        
# 1.类名()自动执行__init__
obj = Foo(1,2)
# 2.对象()自动执行__call__
obj(4,3,2,k1=234)
# 3.对象[]自动执行__getitem__
ret = obj['yu']
print(ret)
# 4.对象['xx'] = 11自动执行__setitem__
obj['k1'] = 123
# 5.del 对象[xx] 自动执行__delitem__
del obj['uuu']
# 6.对象+对象  自动执行__add__
obj1 = Foo(1,2)
obj2 = Foo(22,33)
ret = obj1 + obj2
print(ret)
# 7.真正的构造方法
Foo(1,2)
```
